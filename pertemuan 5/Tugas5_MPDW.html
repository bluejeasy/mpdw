Plot Box-Cox

library(MASS)

# Index untuk subset data (115 data: 461575)
index <- seq(1:nrow(subset_data))

# Buat model regresi sederhana (value ~ index)
model <- lm(subset_data$IPN31152N ~ index)

# Box-Cox dengan rentang lambda -1 sampai 4
bc <- boxcox(
  model,
  lambda = seq(-1, 4, by = 0.01),
  main = "Box-Cox Transformation untuk IPN31152N"
)
## Warning: In lm.fit(x, y, offset = offset, singular.ok = singular.ok, ...) :
##  extra argument 'main' will be disregarded

# Nilai lambda optimum
lambda <- bc$x[which.max(bc$y)]
lambda
## [1] 1.6
# Selang kepercayaan 95% untuk lambda
ci_lambda <- bc$x[bc$y > max(bc$y) - 0.5 * qchisq(0.95, 1)]

# Batas bawah dan batas atas
ci_lower <- min(ci_lambda)
ci_upper <- max(ci_lambda)

c(Batas_Bawah = ci_lower, Batas_Atas = ci_upper)
## Batas_Bawah  Batas_Atas 
##        0.52        2.69

Gambar di atas menunjukkan nilai optimum sebesar 1,6 dengan selang kepercayaan 95% pada rentang (0,52 ; 2,69). Selang kepercayaan tersebut memuat nilai 1 sehingga dapat disimpulkan bahwa data stasioner dalam ragam.
